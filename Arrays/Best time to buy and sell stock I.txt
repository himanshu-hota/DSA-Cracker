public int maxProfit(int[] prices) {
        
        int buy = prices[0]; // buy stock at day first for initial stage
        int maxProfit = 0;  // Set maxprofit to 0 for initial
        int n= prices.length;
        for(int i =1;i<n;i++){  //loop through day 2 to day end
            if(prices[i]<buy)   // if current day price is lower than any previous day then buy stock today
                buy = prices[i]; 
            int profit = prices[i]-buy;  // calculate profit
            
            if(profit>maxProfit)	//if current profit is greater than privious profit then update it
                maxProfit = profit;
        }
        
        return maxProfit;   //return maximum profit
    }
