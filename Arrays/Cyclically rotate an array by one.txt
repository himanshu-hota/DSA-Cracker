  public void rotate(int arr[], int n)
    {
        int x = arr[n-1]; // store last element 
        
        for(int i =arr.length-1 ;i>0;i--){
           arr[i] = arr[i-1]; //Shift Array element by one element
        }
        
        arr[0] = x; //insert last element to first
        
    }
