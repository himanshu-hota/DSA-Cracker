class Solution {
    public void reverseString(char[] s) {
        
        reverse(s, 0 , s.length-1);
        
    }
    
    public static void reverse(char[] str , int start , int end){
        //Base case
        if(start>=end)
            return;
        
        //Swap values
        char temp = str[start];
        str[start] = str[end];
        str[end] = temp;
        
        //Recursive call
        reverse(str,start+1,end-1);
    }
}
